    // فئة اللاعب (Player) - تتضمن الفيزياء الحديثة
    // ==========================================
    class Player extends Phaser.Physics.Arcade.Sprite {
        constructor(scene, x, y) {
            // نستخدم صورة شخصية افتراضية مؤقتاً (سيتم القص من الشيت المجمع لاحقاً)
            super(scene, x, y, 'characters_sheet'); 
            scene.add.existing(this);
            scene.physics.add.existing(this);

            this.setCollideWorldBounds(true);
            this.setOrigin(0.5, 1);
            
            // خصائص الحركة والفيزياء الحديثة
            this.speed = 250;
            this.jumpForce = -550;
            
            // ميزات اللعب الحديث (Coyote Time & Jump Buffer) لجعل القفز سلساً وعادلاً
            this.coyoteTime = 150; // يسمح بالقفز لكسر من الثانية بعد مغادرة المنصة
            this.jumpBuffer = 150; // يحفظ ضغطة القفز لكسر من الثانية قبل ملامسة الأرض
            this.lastGroundedTime = 0;
            this.lastJumpPressedTime = 0;

            // إحصائيات اللاعب
            this.health = 3;
            this.hasShield = false;
            this.isInvulnerable = false; // مناعة مؤقتة بعد التضرر
            this.coins = 0;

            this.scene = scene;
        }

        update(time, cursors, mobileInput) {
            // التحقق مما إذا كان اللاعب يلامس الأرض
            const onGround = this.body.touching.down || this.body.blocked.down;
            
            if (onGround) {
                this.lastGroundedTime = time;
            }

            // الحركة الأفقية (تعمل مع لوحة المفاتيح أو شاشة اللمس)
            let isMovingLeft = cursors.left.isDown || (mobileInput && mobileInput.left);
            let isMovingRight = cursors.right.isDown || (mobileInput && mobileInput.right);

            if (isMovingLeft) {
                this.setVelocityX(-this.speed);
                // قلب اتجاه الشخصية
                this.setFlipX(true); 
            } else if (isMovingRight) {
                this.setVelocityX(this.speed);
                this.setFlipX(false);
            } else {
                this.setVelocityX(0);
            }

            // القفز
            let jumpPressed = Phaser.Input.Keyboard.JustDown(cursors.up) || (mobileInput && mobileInput.jumpPressed);
            
            if (jumpPressed) {
                this.lastJumpPressedTime = time;
                if(mobileInput) mobileInput.jumpPressed = false; // تصفير زر الهاتف
            }

            // التحقق من إمكانية القفز بناءً على الفيزياء الحديثة
            let canJump = (time - this.lastGroundedTime <= this.coyoteTime);
            let wantsToJump = (time - this.lastJumpPressedTime <= this.jumpBuffer);

            if (wantsToJump && canJump) {
                this.setVelocityY(this.jumpForce);
                this.lastJumpPressedTime = 0; // استهلاك القفزة
                this.lastGroundedTime = 0;
                
                // تأثير الغبار والصوت (Game Juice)
                this.createDust();
                this.scene.sound.play('jump', { volume: 0.5 });
            }
        }

        takeDamage() {
            if (this.isInvulnerable) return false;

            // إذا كان يمتلك درعاً، يخسره بدلاً من الصحة
            if (this.hasShield) {
                this.hasShield = false;
                this.triggerInvulnerability();
                this.scene.cameras.main.shake(100, 0.01); // اهتزاز خفيف للشاشة
                return false; 
            }

            // خسارة الصحة
            this.health--;
            this.triggerInvulnerability();
            this.scene.cameras.main.shake(250, 0.02); // اهتزاز أقوى
            
            if (this.health <= 0) {
                this.die();
            }
            return true;
        }

        triggerInvulnerability() {
            this.isInvulnerable = true;
            this.setAlpha(0.5); // جعل الشخصية شفافة قليلاً
            this.scene.time.delayedCall(1500, () => {
                this.isInvulnerable = false;
                this.setAlpha(1);
            });
        }

        die() {
            // سنقوم لاحقاً بربط هذا بدالة إعادة الإحياء في المستوى
            if(this.scene.respawnPlayer) {
                this.scene.respawnPlayer();
            }
        }

        createDust() {
            // إنشاء تأثير جزيئات الغبار عند القفز
            let dust = this.scene.add.rectangle(this.x, this.y, 8, 8, 0xffffff, 0.7);
            this.scene.tweens.add({
                targets: dust,
                y: this.y + 15,
                x: this.x + (Math.random() * 10 - 5),
                alpha: 0,
                scale: 1.5,
                duration: 400,
                onComplete: () => dust.destroy()
            });
        }
    }

    // فئة الأعداء (Enemy) - ذكاء اصطناعي متطور
    // ==========================================
    class Enemy extends Phaser.Physics.Arcade.Sprite {
        constructor(scene, x, y, level) {
            super(scene, x, y, 'characters_sheet');
            scene.add.existing(this);
            scene.physics.add.existing(this);
            
            this.setCollideWorldBounds(true);
            this.level = level;
            this.speed = 100 + (level * 10); // سرعة تزداد مع تقدم المستويات
            this.setVelocityX(-this.speed);
            
            // خصائص الرماية (للمستويات المتقدمة)
            this.canShoot = level >= 7;
            this.lastShotTime = 0;
        }

        update(time) {
            // انعكاس الحركة عند الاصطدام بحائط أو حافة
            if (this.body.blocked.left) {
                this.setVelocityX(this.speed);
                this.setFlipX(true);
            } else if (this.body.blocked.right) {
                this.setVelocityX(-this.speed);
                this.setFlipX(false);
            }

            // ذكاء الرماية في مستويات الحرب
            if (this.canShoot && time > this.lastShotTime + 2000) {
                this.shoot();
                this.lastShotTime = time;
            }
        }

        shoot() {
            let bullet = this.scene.physics.add.sprite(this.x, this.y, 'characters_sheet');
            bullet.body.allowGravity = false;
            let direction = this.flipX ? 1 : -1;
            bullet.setVelocityX(300 * direction);
            this.scene.enemyBullets.add(bullet);
            // تدمير الرصاصة بعد فترة
            this.scene.time.delayedCall(2000, () => bullet.destroy());
        }
    }

    // نظام صناديق الأسئلة (Question Box System)
    // ==========================================
    class QuestionBox extends Phaser.Physics.Arcade.Sprite {
        constructor(scene, x, y) {
            super(scene, x, y, 'characters_sheet'); // سنستخدم جزءاً معيناً من الصورة لاحقاً
            scene.add.existing(this);
            scene.physics.add.existing(this, true); // جسم ساكن (Static)
            this.setTint(0xffff00); // تلوين الصندوق بالذهبي
            this.isOpened = false;
        }

        open() {
            if (this.isOpened) return;
            this.isOpened = true;
            this.setAlpha(0.5);
            
            // تعطيل فيزياء اللعبة مؤقتاً لعرض السؤال
            this.scene.physics.world.pause();
            this.showQuestionUI();
        }

        showQuestionUI() {
            const scene = this.scene;
            const width = scene.cameras.main.width;
            const height = scene.cameras.main.height;

            // خلفية السؤال
            let overlay = scene.add.rectangle(width/2, height/2, 500, 350, 0x000000, 0.9).setScrollFactor(0).setDepth(100);
            let qText = scene.add.text(width/2, height/2 - 100, 'سؤال تاريخي: ما هو عام الثورة؟', { 
                fontSize: '24px', fill: '#fff', align: 'center', wordWrap: { width: 400 } 
            }).setOrigin(0.5).setScrollFactor(0).setDepth(101);

            // الخيارات
            let options = ['1917', '1945', '1991'];
            options.forEach((opt, index) => {
                let btn = scene.add.text(width/2, height/2 + (index * 50), opt, { 
                    fontSize: '22px', fill: '#ffff00', backgroundColor: '#333', padding: 10 
                }).setOrigin(0.5).setScrollFactor(0).setDepth(101).setInteractive();

                btn.on('pointerdown', () => {
                    if (opt === '1917') {
                        scene.player.coins += 10;
                        scene.player.hasShield = true;
                        // تأثير صوت النجاح
                    }
                    // إغلاق الواجهة واستئناف اللعب
                    overlay.destroy();
                    qText.destroy();
                    scene.children.list.filter(c => c.depth === 101 || c.depth === 100).forEach(c => c.destroy());
                    scene.physics.world.resume();
                });
            });
        }
    }

    // المشهد الأساسي للمستويات (BaseLevelScene)

    class BaseLevelScene extends Phaser.Scene {
        constructor(key, levelNumber, bgIndex) {
            super({ key: key });
            this.levelNumber = levelNumber;
            this.bgIndex = bgIndex; // رقم الخلفية من الصورة المجمعة (1-8)
        }

        create() {
            const width = this.scale.width;
            const height = this.scale.height;

            // 1. إعداد الخلفية (تقطيع من الصورة المجمعة 710bc8.jpg)
            // بما أن الصورة تحتوي على 8 خلفيات (2 أعمدة و4 صفوف)
            this.setupBackground();

            // 2. إنشاء المجموعات (Groups) لسهولة الإدارة
            this.platforms = this.physics.add.staticGroup();
            this.enemies = this.physics.add.group({ runChildUpdate: true });
            this.coins = this.physics.add.staticGroup();
            this.questionBoxes = this.physics.add.staticGroup();
            this.enemyBullets = this.physics.add.group();

            // 3. إنشاء اللاعب
            this.player = new Player(this, 100, height - 150);
            
            // 4. إعداد واجهة المستخدم والتحكم
            this.ui = new GameUI(this);
            this.cursors = this.input.keyboard.createCursorKeys();

            // 5. إعداد الكاميرا الحديثة
            this.cameras.main.setBounds(0, 0, 5000, height); // مستوى طويل جداً
            this.physics.world.setBounds(0, 0, 5000, height);
            this.cameras.main.startFollow(this.player, true, 0.1, 0.1);
            this.cameras.main.setLerp(0.1, 0.1); // حركة كاميرا ناعمة

            // 6. نظام التصادم (Collisions)
            this.physics.add.collider(this.player, this.platforms);
            this.physics.add.collider(this.enemies, this.platforms);
            
            // التفاعل مع العملات
            this.physics.add.overlap(this.player, this.coins, (p, coin) => {
                coin.destroy();
                this.player.coins += 1;
                this.sound.play('jump', { volume: 0.2, detune: 500 }); // صوت جمع
            });

            // التفاعل مع الصناديق
            this.physics.add.collider(this.player, this.questionBoxes, (p, box) => {
                if (p.body.touching.up && box.body.touching.down) {
                    box.open();
                }
            });

            // التصادم مع الأعداء
            this.physics.add.overlap(this.player, this.enemies, (p, enemy) => {
                if (p.body.touching.down && enemy.body.touching.up) {
                    // قتل العدو عند القفز عليه
                    enemy.destroy();
                    p.setVelocityY(-400); // قفزة ارتدادية
                } else {
                    p.takeDamage();
                }
            });

            // إنشاء تضاريس أولية (أرضية) - سنخصصها في كل مستوى
            this.createFloor(5000);
        }

        setupBackground() {
            // حساب إحداثيات الخلفية بناءً على الرقم bgIndex
            // الصورة 710bc8.jpg مقسمة 2 عرضاً و 4 طولاً
            let col = (this.bgIndex - 1) % 2;
            let row = Math.floor((this.bgIndex - 1) / 2);
            
            // إضافة الخلفية كـ TileSprite لدعم التكرار (Parallax)
            let bg = this.add.tileSprite(0, 0, 5000, 720, 'backgrounds_sheet')
                .setOrigin(0, 0)
                .setScrollFactor(0.5); // تأثير العمق (Parallax)
            
            // تحديد الجزء المقطوع من الصورة (هذا الجزء تقريبي ويحتاج لضبط حسب أبعاد الصورة الأصلية)
            bg.setTilePosition(col * 640, row * 360); 
        }

        createFloor(length) {
            for (let i = 0; i < length; i += 128) {
                let ground = this.platforms.create(i, 700, 'characters_sheet').setScale(2).refreshBody();
                ground.setTint(0x333333); // جعل الأرضية داكنة لتبدو كمنصة
                ground.setVisible(false); // سنستبدلها لاحقاً برسومات أفضل
                
                // إضافة خط أرضي مرئي
                this.add.rectangle(i, 710, 128, 20, 0x222222).setOrigin(0, 0);
            }
        }

        respawnPlayer() {
            // إعادة اللاعب إلى بداية المستوى أو أقرب نقطة (Checkpoint)
            this.player.setPosition(100, 400);
            this.player.health = 3;
            this.cameras.main.flash(500, 255, 0, 0);
        }

        update(time, delta) {
            this.player.update(time, this.cursors, this.ui.mobileInput);
            this.ui.updateHUD(this.player.health, this.player.coins, this.player.hasShield);

            // تحديث الأعداء
            this.enemies.getChildren().forEach(enemy => enemy.update(time));
        }
    }// ==========================================

    // المستوى 8: المرحلة النهائية والزعيم (1945 - 1953)
    class Level8 extends BaseLevelScene {
        constructor() { super('Level8', 8, 8); }
        create() {
            super.create();
            this.add.text(400, 300, 'المستوى 8: المواجهة النهائية', { fontSize: '32px', fill: '#fff' }).setOrigin(0.5).setScrollFactor(0.2);
            
            // إنشاء الزعيم النهائي (Boss)
            this.boss = this.physics.add.sprite(4500, 500, 'characters_sheet').setScale(4).setTint(0xff0000);
            this.boss.health = 10;
            this.boss.body.allowGravity = false;
            this.boss.setImmovable(true);

            // نمط هجوم الزعيم
            this.time.addEvent({
                delay: 2000,
                callback: () => {
                    if (this.boss.active) {
                        let b = this.physics.add.sprite(this.boss.x, this.boss.y, 'characters_sheet').setTint(0xff0000);
                        this.physics.moveToObject(b, this.player, 300);
                        this.enemyBullets.add(b);
                    }
                },
                loop: true
            });

            // هزيمة الزعيم عند القفز عليه 10 مرات
            this.physics.add.overlap(this.player, this.boss, (p, b) => {
                if (p.body.touching.down && b.body.touching.up) {
                    b.health--;
                    p.setVelocityY(-500);
                    this.cameras.main.shake(200, 0.03);
                    if (b.health <= 0) {
                        b.destroy();
                        this.add.text(width/2, height/2, 'لقد انتصرت! النهاية', { fontSize: '64px', fill: '#ffd700' }).setOrigin(0.5).setScrollFactor(0);
                    }
                } else {
                    p.takeDamage();
                }
            });
        }
    }// ==========================================
