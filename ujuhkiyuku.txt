let audioCtx = null;
let masterVol = 0.45;

function getACtx() {
  if (!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)();
  return audioCtx;
}
function playTone(freq, dur, type='square', vol=0.25, delay=0) {
  if (masterVol === 0) return;
  try {
    const ctx = getACtx(), osc = ctx.createOscillator(), g = ctx.createGain();
    osc.connect(g); g.connect(ctx.destination);
    osc.type = type;
    osc.frequency.setValueAtTime(freq, ctx.currentTime + delay);
    g.gain.setValueAtTime(0, ctx.currentTime + delay);
    g.gain.linearRampToValueAtTime(vol * masterVol, ctx.currentTime + delay + 0.01);
    g.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + delay + dur);
    osc.start(ctx.currentTime + delay);
    osc.stop(ctx.currentTime + delay + dur + 0.01);
  } catch(e) {}
}

const QUIZ_TXT = {

const OPT_COLORS = ['hsl(330,65%,65%)','hsl(25,85%,60%)','hsl(38,75%,60%)','hsl(200,60%,60%)'];
const OPT_LETTERS = ['A','B','C','D'];

function initQuiz() {
  const el = document.getElementById('screen-quiz');
  el.dir = state.dir;
  document.getElementById('quiz-user-name').textContent = state.userName;
  state.currentQ = 0;
  state.answers = {};
  renderQuestion();
}
function renderQuestion() {
  const t = QUIZ_TXT[state.lang];
  const i = state.currentQ;
  const q = QUIZ_QUESTIONS[i];
  const total = QUIZ_QUESTIONS.length;
  const prog = (i / total) * 100;

  document.getElementById('q-label').textContent       = t.label;
  document.getElementById('q-num').textContent         = String(i+1).padStart(2,'0') + ' / ' + String(total).padStart(2,'0');
  document.getElementById('quiz-title-label').textContent = t.title;
  document.getElementById('q-badge-num').textContent   = 'Q' + (i+1);
  document.getElementById('q-text').textContent        = q.q[state.lang];
  document.getElementById('quiz-progress-fill').style.width = prog + '%';
  document.getElementById('quiz-pct').textContent      = Math.round(prog) + '%';

  const sel = state.answers[i];
  const isLast = i === total - 1;
  document.getElementById('btn-next').textContent = isLast ? t.submit : t.next;
  document.getElementById('btn-next').disabled    = (sel === undefined);
  document.getElementById('btn-prev').disabled    = (i === 0);
  document.getElementById('prev-btn') && (document.getElementById('btn-prev').querySelector('span') || document.getElementById('btn-prev')).textContent;

  const opts = document.getElementById('options-container');
  opts.innerHTML = '';
  q.opts.forEach((o, oi) => {
    const isSelected = (sel === oi);
    const c = OPT_COLORS[oi];
    const btn = document.createElement('button');
    btn.className = 'option-btn' + (isSelected ? ' selected' : '');
    btn.style.cssText = isSelected ? `border:2px solid ${c} !important;background:${c}22 !important;` : '';
    btn.innerHTML = `
      <span class="option-letter" style="${isSelected?`background:${c}33;border-color:${c};color:${c}`:`color:rgba(180,160,140,0.6)`}">${OPT_LETTERS[oi]}</span>
      <span style="flex:1">${o[state.lang]}</span>
      ${isSelected ? `<span style="color:${c};">▶</span>` : ''}`;
    btn.onclick = () => selectOption(oi);
    opts.appendChild(btn);
  });
}
function selectOption(oi) {
  sndSelect();
  state.answers[state.currentQ] = oi;
  renderQuestion();
}
function nextQ() {
  if (state.answers[state.currentQ] === undefined) return;
  sndClick();
  if (state.currentQ === QUIZ_QUESTIONS.length - 1) {
    finishQuiz(); return;
  }
  state.currentQ++;
  renderQuestion();
}
function prevQ() {
  if (state.currentQ === 0) return;
  sndClick();
  state.currentQ--;
  renderQuestion();
}
function finishQuiz() {
  sndSuccess();
  showScreen('analyzing');
  // Calculate scores
  const scores = { thabet:0, kawthar:0, malak:0, mohamed:0 };
  QUIZ_QUESTIONS.forEach((q, i) => {
    const oi = state.answers[i];
    if (oi !== undefined) {
      const s = q.opts[oi].s;
      scores.thabet   += s.thabet;
      scores.kawthar  += s.kawthar;
      scores.malak    += s.malak;
      scores.mohamed  += s.mohamed;
    }
  });
  state.resultScores = scores;
  const winner = Object.entries(scores).reduce((a,b) => a[1]>b[1]?a:b)[0];
  state.resultChar = CHARACTERS[winner];

  // Animated progress
  let p = 0;
  const sp = document.getElementById('a-bar');
  const pp = document.getElementById('a-pct');
  const c = state.resultChar.accent;
  sp.style.background = c; pp.style.color = c;
  document.getElementById('analyze-spinner').style.borderTopColor = c;
  const title = state.lang === 'ar' ? 'جارٍ تحليل إجاباتك...' : 'Analyzing your responses...';
  document.getElementById('analyze-title').textContent = title;

  const iv = setInterval(()=>{
    p = Math.min(p + 1.4, 100);
    sp.style.width = p + '%';
    pp.textContent = Math.round(p) + '%';
    if (p >= 100) {
      clearInterval(iv);
      sndTransition();
      setTimeout(()=>{ showScreen('reveal'); buildReveal(); spawnParticles(); }, 300);
    }
  }, 30);
}

function initFaceMesh() {
  faceMesh = new FaceMesh({
    locateFile: f => `https://cdn.jsdelivr.net/npm/@mediapipe/face_mesh/${f}`
  });
  faceMesh.setOptions({ maxNumFaces:1, refineLandmarks:true, minDetectionConfidence:0.6, minTrackingConfidence:0.6 });
  faceMesh.onResults(res => {
    faceRes = (res.multiFaceLandmarks?.length > 0) ? res.multiFaceLandmarks[0] : null;
  });
}

/* ════════════════════════════════════════════
   18. CAMERA
════════════════════════════════════════════ */
function startCamera() {
  const video  = document.getElementById('input_video');
  const canvas = document.getElementById('output_canvas');
  const ctx    = canvas.getContext('2d');

  cameraInstance = new Camera(video, {
    onFrame: async () => {
      await faceMesh.send({ image: video });
      canvas.width  = video.videoWidth  || 640;
      canvas.height = video.videoHeight || 480;
      ctx.save();
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      ctx.drawImage(video, 0, 0, canvas.width, canvas.height);
      if (faceRes) processFrame(faceRes);
      ctx.restore();
    },
    width: 640, height: 480
  });
  cameraInstance.start()
    .then(()=>{ document.getElementById('cam-error').style.display = 'none'; })
    .catch(()=>{ document.getElementById('cam-error').style.display = 'flex'; document.getElementById('cal-overlay').style.display = 'none'; });
}
function retryCamera() {
  document.getElementById('cam-error').style.display = 'none';
  startCamera();
}

/* ════════════════════════════════════════════
   19. FACE ANALYSIS
════════════════════════════════════════════ */
function processFrame(lm) {
  const now  = Date.now();
  const faceW = dist(lm[234], lm[454]);
  if (faceW < 0.001) return;

  const cEAR         = ((dist(lm[159],lm[145]) + dist(lm[386],lm[374])) / 2) / faceW;
  const cMouth       = dist(lm[13],  lm[14])  / faceW;
  const cInnerBrow   = dist(lm[55],  lm[285]) / faceW;
  const cOuterBrow   = ((dist(lm[105],lm[159]) + dist(lm[334],lm[386])) / 2) / faceW;
  const cNoseWrinkle = dist(lm[1],   lm[168]) / faceW;

  if (calPaused) return;

  // ─── Calibration ───────────────────────
  if (!isCalibrated) {
    calFrames.push({ ear:cEAR, mouth:cMouth, innerBrow:cInnerBrow, outerBrow:cOuterBrow, noseWrinkle:cNoseWrinkle });
    const pct = Math.round((calFrames.length / CAL_FRAMES) * 100);
    document.getElementById('cal-bar-fill').style.width = pct + '%';
    const l = state.lang;
    document.getElementById('cal-status').textContent = pct + (l==='ar'?'% مكتمل':'% complete');
    if (calFrames.length >= CAL_FRAMES) {
      const avg = k => calFrames.reduce((s,f)=>s+f[k],0) / calFrames.length;
      baseFace = { ear:avg('ear'), mouth:avg('mouth'), innerBrow:avg('innerBrow'), outerBrow:avg('outerBrow'), noseWrinkle:avg('noseWrinkle') };
      isCalibrated = true; lastRecalTime = now;
      document.getElementById('cal-overlay').style.display = 'none';
      document.getElementById('mental-state').textContent = CHARACTERS[state.resultChar.key].msgShort[state.lang];
    }
    return;
  }

  // Recalibrate every 5 min
  if (now - lastRecalTime > 300000) { lastRecalTime = now; startCalibration(); return; }

  // ─── Sleep detection ───────────────────
  const eyesClosed = cEAR < baseFace.ear * 0.55;
  if (eyesClosed) {
    if (sleepStart === 0) sleepStart = now;
    if (now - sleepStart >= DURATIONS.sleep && !isSleeping) { isSleeping = true; activateSleep(); }
  } else {
    sleepStart = 0;
    if (isSleeping) { isSleeping = false; sleepScored = false; }
  }
  if (isSleeping) return;

  // ─── Mouth analysis ────────────────────
  mouthHist.push({ time:now, val:cMouth });
  mouthHist = mouthHist.filter(m => now - m.time < 1500);
  let changes = 0;
  for (let i = 1; i < mouthHist.length; i++)
    if (Math.abs(mouthHist[i].val - mouthHist[i-1].val) > 0.006) changes++;
  const isTalking   = changes > 6;
  const isMouthOpen = cMouth > baseFace.mouth * 2.8;
  const isYawn      = isMouthOpen && !isTalking && cEAR < baseFace.ear * 0.88;
  let mouthLabel    = state.lang === 'ar' ? 'صامت 😐' : 'Silent 😐';

  if (isYawn) {
    if (mouthOpenStart === 0) mouthOpenStart = now;
    const dur = now - mouthOpenStart;
    if (dur >= DURATIONS.fatigue) {
      mouthLabel = state.lang === 'ar' ? 'تثاؤب 🥱' : 'Yawning 🥱';
      if (!yawnScored && now - lastAction.fatigue >= ACTION_CD) {
        awardScore('fatigue'); sessionStats.fatigue++; cumStats.fatigue++;
        yawnScored = true; lastAction.fatigue = now;
      }
    } else {
      const s = Math.ceil((DURATIONS.fatigue - dur) / 1000);
      mouthLabel = (state.lang==='ar'?`فم مفتوح... (${s}ث)`:`Mouth open... (${s}s)`);
    }
  } else {
    if (mouthOpenStart !== 0) yawnScored = false;
    mouthOpenStart = 0;
    if (isTalking) {
      mouthLabel = state.lang === 'ar' ? 'يتحدث 🗣️' : 'Talking 🗣️';
      if (now - lastTalkTime >= 3000) {
        awardScore('talking'); sessionStats.talking++; cumStats.talking++;
        lastTalkTime = now;
      }
    } else if (cMouth > baseFace.mouth * 1.3) {
      mouthLabel = state.lang === 'ar' ? 'شفاه متباعدة' : 'Mouth ajar';
    }
  }

  // ─── Blink analysis ────────────────────
  if (cEAR < baseFace.ear * 0.72) {
    if (blinkStart === 0) blinkStart = now;
    isClosed = true;
  } else {
    if (isClosed && blinkStart !== 0) {
      const dur = now - blinkStart;
      if (dur >= 500 && dur < 2000 && now - lastAction.understanding >= ACTION_CD) {
        awardScore('understanding'); sessionStats.understanding++; cumStats.understanding++;
        lastAction.understanding = now;
      } else { blinks++; }
    }
    blinkStart = 0; isClosed = false;
  }

  // ─── Brow analysis ─────────────────────
  let detBrow = "none", browLabel = state.lang==='ar'?'طبيعي':'Normal';
  if (cOuterBrow > baseFace.outerBrow * 1.18) {
    detBrow = 'surprise'; browLabel = state.lang==='ar'?'تعجب 😲':'Surprise 😲';
  } else if (cInnerBrow < baseFace.innerBrow * 0.93 && cNoseWrinkle < baseFace.noseWrinkle * 0.93) {
    detBrow = 'confusion'; browLabel = state.lang==='ar'?'ارتباك 🧐':'Confused 🧐';
  } else if (cInnerBrow < baseFace.innerBrow * 0.93) {
    detBrow = 'concentration'; browLabel = state.lang==='ar'?'تركيز 🔥':'Focus 🔥';
  }
  if (detBrow !== browState) { browState = detBrow; browStart = now; browScored = false; }
  if (!browScored && detBrow !== 'none') {
    const elapsed = now - browStart;
    const minDur  = DURATIONS[detBrow] || 1500;
    if (elapsed >= minDur && now - lastAction[detBrow] >= ACTION_CD) {
      awardScore(detBrow);
      if (detBrow === 'concentration') sessionStats.concentration++;
      if (detBrow === 'confusion')     sessionStats.confusion++;
      if (detBrow === 'surprise')      { sessionStats.surprise++; cumStats.surprise++; }
      lastAction[detBrow] = now; browScored = true;
    }
  }

  // ─── Update UI ─────────────────────────
  document.getElementById('total-score-val').textContent = Math.floor(totalScore);
  document.getElementById('blink-val').textContent    = cumStats.understanding;
  document.getElementById('talk-val').textContent     = cumStats.talking;
  document.getElementById('surprise-val').textContent = cumStats.surprise;
  document.getElementById('fatigue-val').textContent  = cumStats.fatigue;
  document.getElementById('brow-val').textContent     = browLabel;
  document.getElementById('mouth-val').textContent    = mouthLabel;

  // Mode check every 2 min, report every 30s
  if (now - lastModeCheck >= 120000) { lastModeCheck = now; checkScoreMode(); }
  if (now - lastReport >= 30000)     { lastReport = now;    processCycle(); }
}

function awardScore(action) {
  totalScore += (CHARACTERS[state.resultChar.key].scores[action] || 0);
}
function activateSleep() {
  document.getElementById('output_canvas').style.filter = 'blur(8px) brightness(0.4)';
  document.getElementById('sleep-overlay').style.display = 'flex';
  if (!sleepScored) { awardScore('sleep'); sleepScored = true; }
  showPopup(state.lang==='ar' ? 'انتبه! يبدو أنك نمت 😴 أكمل درسك!' : 'Wake up! You fell asleep 😴 Continue your lesson!');
}
function dismissSleep() {
  isSleeping = false; sleepScored = false; sleepStart = 0;
  document.getElementById('output_canvas').style.filter = '';
  document.getElementById('sleep-overlay').style.display = 'none';
}
function showPopup(msg) {
  const cfg  = CHARACTERS[state.resultChar.key];
  const l    = state.lang;
  document.getElementById('pop-name').textContent = cfg.name[l];
  document.getElementById('pop-name').style.color = cfg.accent;
  document.getElementById('pop-msg').textContent  = msg;
  const pop = document.getElementById('char-popup');
  pop.style.borderColor = cfg.accent;
  pop.style.display = 'block';
  clearTimeout(pop._timer);
  pop._timer = setTimeout(()=> pop.style.display = 'none', 6000);
}

function checkScoreMode() {
  const s = totalScore;
  if      (s < -20)  RescueMode();
  else if (s <= 50)  GuidanceMode();
  else if (s <= 200) BoostMode();
  else               MasteryMode();
}
function processCycle() {
  const l = state.lang;
  let msg = '';
  if (sessionStats.fatigue > 0)
    msg = l==='ar' ? 'لاحظت تعباً، هل تحتاج لاستراحة قصيرة؟ 😴' : 'I noticed fatigue. Would you like a short break? 😴';
  else if (sessionStats.understanding > 0)
    msg = l==='ar' ? 'رائع! الأفكار تترسّخ في ذهنك ✨' : 'Great! The ideas are settling in your mind ✨';
  else if (sessionStats.concentration > 2)
    msg = l==='ar' ? 'تركيز ممتاز! أنت على المسار الصحيح 🎯' : 'Excellent focus! You are on the right track 🎯';
  if (msg) showPopup(msg);
  sessionStats = { concentration:0, confusion:0, fatigue:0, understanding:0, surprise:0, talking:0 };
}

  document.getElementById('session-end').style.display = 'flex';
  const l = state.lang;
  document.getElementById('end-title').textContent        = l==='ar' ? 'انتهت جلسة التعلم!' : 'Learning Session Complete!';
  document.getElementById('final-score-display').textContent = Math.floor(totalScore);
  const msg = totalScore > 100 ? (l==='ar'?'أداء استثنائي! أنت نجم اليوم 🏆':'Exceptional performance! You are today\'s star 🏆')
            : totalScore > 0   ? (l==='ar'?'أداء جيد، استمر في التقدم 📚':'Good performance, keep progressing 📚')
            :                    (l==='ar'?'الغد فرصة جديدة للتميز 🌱':'Tomorrow is a new chance to shine 🌱');
  document.getElementById('end-subtitle').textContent = msg;
}

  if (cameraInstance) { try { cameraInstance.stop(); } catch(e){} cameraInstance = null; }

  Object.assign(state, { lang:'ar', dir:'rtl', userName:'', answers:{}, currentQ:0, resultChar:null, resultScores:{}, checks:{c1:false,c2:false,c3:false} });

  mouthOpenStart = 0; blinkStart = 0; sleepStart = 0; lastTalkTime = 0;
  mouthHist = []; cumStats = { understanding:0, talking:0, surprise:0, fatigue:0 };
  sessionStats = { concentration:0, confusion:0, fatigue:0, understanding:0, surprise:0, talking:0 };
  document.documentElement.style.setProperty('--accent', '#C96482');
  document.getElementById('screen-dashboard').classList.remove('active');
  document.getElementById('session-end').style.display = 'none';
  document.getElementById('output_canvas').style.filter = '';
  location.reload();
}
